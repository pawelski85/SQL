SELECT U.NAME, P.DESCRIPTION
FROM ISSUESLISTS U JOIN ISSUES P ON P.ISSUESLIST_ID=U.ID
ORDER BY U.NAME, P.DESCRIPTION;


2.
SELECT P.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM USERS U JOIN ISSUES P ON P.USER_ID_ASSIGNEDTO=U.ID;


3.
SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES
FROM USERS U JOIN ISSUES P ON P.USER_ID_ASSIGNEDTO=U.ID
GROUP BY U.ID
HAVING COUNT(*)>1
ORDER BY U.FIRSTNAME, U.LASTNAME;
